const searchkeyword = (e) => {
    e.preventDefault();
    const keyword = document.querySelector(".search").value;
    const API_KEY = "AIzaSyDOdK4QlBWgygR9tPyPJU6fDryHjwlOjyc";
    const apiUrl = `https://www.googleapis.com/youtube/v3/search?q=${encodeURIComponent(keyword)}&key=${API_KEY}`;

    fetch(apiUrl)
        .then(response => response.json())
        .then(data => {
            const search_appearances = data.pageInfo.totalResults;
            const outputElement = document.getElementById("search-output");
            outputElement.innerHTML = `"${keyword}" appeared in ${search_appearances} search results.`;
        })
        .catch(error => {
            console.error('An error occurred:', error);
        });
};

document.getElementById("search-form").addEventListener("submit", searchkeyword);
